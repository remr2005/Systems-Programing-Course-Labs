
; Место начала массива
LXI H, ARRAY
; В  регистре B буду хранить размер массива
mvi b, 7
; В регистре C буду хранить число с которым будет проходить сравнение
mvi c, 25h

mov d, b
mov a, c


LOOP:
mov e, m
cmp M
jnc LABEL
DCR d ; Буду производить декремент от общего кол-во элементов, каждый раз когда буду находить число большее данного
LABEL:
inx h
dcr b
jnz LOOP

hlt
ARRAY:
DB 05H
DB 0AH
DB 1CH
DB FFH
DB 80H
DB 33H
DB 77H
